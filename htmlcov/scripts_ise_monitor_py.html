128def sync_ise(): 

129 log = [] 

130 msg = "" 

131 append_log(log, "ise_monitor::sync_ise::Checking ISE Accounts for re-sync...") 

132 

133 # Ensure that Meraki Dashboard has already completed a sync if it is the source of truth 

134 stat = SyncSession.objects.filter(Q(ise_source=True) | 

135 (Q(dashboard__last_sync__isnull=False) & 

136 Q(iseserver__last_sync__isnull=True)) | 

137 (Q(dashboard__last_sync__isnull=False) & 

138 Q(dashboard__last_sync__gte=F('iseserver__last_sync')))) 

139 

140 if len(stat) <= 0: 

141 append_log(log, "ise_monitor::sync_ise::Skipping sync as Meraki is primary and hasn't synced.") 

142 msg = "SYNC_ISE-DASHBOARD_NEEDS_SYNC" 

143 else: 

144 append_log(log, "ise_monitor::sync_ise::Running sync") 

145 

146 for s in stat: 

147 ctime = make_aware(datetime.datetime.now()) - datetime.timedelta(seconds=s.sync_interval) 

148 # Perform sync if one of the following conditions is met 

149 # 1) The Sync Session is set to Force Rebuild (this one shouldn't be seen here. but just in case...) 

150 # 2) The Dashboard Instance is set to Force Rebuild 

151 # 3) The timestamp of the ISE Server database object isn't the same as the timestamp of it's last sync 

152 # 4) The timestamp of the ISE Server database object's last sync is beyond the configured manual sync timer 

153 dbs = SyncSession.objects.filter(Q(iseserver__force_rebuild=True) | 

154 Q(force_rebuild=True) | 

155 ~Q(iseserver__last_sync=F('iseserver__last_update')) | 

156 Q(iseserver__last_sync__lte=ctime)) 

157 for d in dbs: 

158 # Log the reason(s) for the current sync 

159 if d.force_rebuild: # pragma: no cover 

160 append_log(log, "ise_monitor::sync_ise::Sync Session Force Rebuild", d) 

161 msg = "SYNC_ISE-SYNCSESSION_FORCE_REBUILD" 

162 d.force_rebuild = False 

163 d.save() 

164 if d.iseserver.force_rebuild: 

165 append_log(log, "ise_monitor::sync_ise::Dashboard Force Rebuild", d) 

166 msg = "SYNC_ISE-ISE_FORCE_REBUILD" 

167 d.iseserver.force_rebuild = False 

168 d.iseserver.save() 

169 if d.iseserver.last_sync and (d.iseserver.last_sync != d.iseserver.last_update): 

170 append_log(log, "ise_monitor::sync_ise::Database Config / Sync Timestamp Mismatch", d) 

171 msg = "SYNC_ISE-CONFIG_SYNC_TIMESTAMP_MISMATCH" 

172 if d.iseserver.last_sync and (d.iseserver.last_sync <= ctime): 

173 append_log(log, "ise_monitor::sync_ise::Past Manual Sync Interval", d) 

174 msg = "SYNC_ISE-PAST_SYNC_INTERVAL" 

175 

176 ingest_ise_data(dbs, log) 

177 

178 digest_database_data(SyncSession.objects.all()[0], log) 

179 

180 append_log(log, "ise_monitor::sync_ise::Done") 

181 db_log("ise_monitor", log) 

182 return msg, log
